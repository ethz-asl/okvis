60 void MultiFrame::resetCameraSystemAndFrames(

62 {
63  cameraSystem_ = cameraSystem;
64  frames_.clear(); // erase -- for safety
65  frames_.resize(cameraSystem.numCameras());
66 
67  // copy cameras
68  for(size_t c = 0; c<numFrames(); ++c){
